00871 void wireComponent( const RuntimeModel::ComponentPtr& component
00872                   , ComponentsSet& leftComponentsToWire
00873                   , SCAWiresSet& availableSCAWires)
00874 {
00875    HLOGGER();   
00876    HLOGGER_INFO(component->uri().c_str());
00877 
00878    //a component that is currently being resolved MUST NOT be resolved twice
00879    assert(component->state() != RuntimeModel::Component::Resolving);  
00880    //a component that is currently being resolved MUST NOT be in Unresolving state 
00881    assert(component->state() != RuntimeModel::Component::Unresolving);
00882    //For now we assume that resolved components must not be processed by wireComponent(). 
00883    //This might change in future. 
00884    assert(component->state() != RuntimeModel::Component::Resolved);
00885 
00886    //We start to resolve the component by trying to wire (=resolve) all of its references.
00887    component->setState(RuntimeModel::Component::Resolving);
00888    
00889    //assume that the component will be in Resolved state an the end
00890    RuntimeModel::Component::State finalComponentState = RuntimeModel::Component::Resolved;
00891    BOOST_FOREACH(const RuntimeModel::ReferenceBasePtr& reference, component->references())
00892    {  
00893       //For now assume that it will be never necessary to wire a reference that is already
00894       //resolved during the resolving process for a whole component. Yes, in case of rewiring
00895       //it might become necessary but in that cast wireReference() or its future 'rewire' 
00896       //counterpart won't be called from here. 
00897       if(reference->state() != RuntimeModel::Reference::Resolved)
00898       {
00899          //***************** TRY TO WIRE THE REFERENCE  *******************
00900          wireReference(reference, leftComponentsToWire, availableSCAWires);
00901          //****************************************************************
00902          
00903          //a reference should definitely have a stable state after wiring
00904          assert(reference->state() != RuntimeModel::Reference::Resolving);
00905          assert(reference->state() != RuntimeModel::Reference::Unresolving);
00906          assert(reference->state() != RuntimeModel::Reference::Initial);
00907          
00908          //If the component's final state has already been set to Unresolved because the wiring 
00909          //of one of the previously processed references didn't succeed, it cannot be changed 
00910          //anymore (see definition for Component::Unresolved)
00911          if(finalComponentState != RuntimeModel::Component::Unresolved)
00912          {  
00913             if(reference->state() == RuntimeModel::Reference::Unresolved)
00914             {
00915                finalComponentState = RuntimeModel::Component::Unresolved;
00916             }
00917             else if(reference->state() == RuntimeModel::Reference::OptionalUnresolved)
00918             {      
00919                finalComponentState = RuntimeModel::Component::OptionalUnresolved;
00920             }
00921          }         
00922       } //if(reference->state() != RuntimeModel::Reference::Resolved)            
00923    } //BOOST_FOREACH(const RuntimeModel::ReferenceBasePtr& reference, component->references())
00924       
00925    //For components implemented by composites, it's not only necessary to wire its references
00926    //(which actually are promoted to further references) but also to wire its implementing
00927    //composite to reach a valid state.
00928    if( component->scaComponent()->implementationElement()->xsdType() == 
00929       XSDTypeImplementationComposite )
00930    {
00931       //If the wiring of the references has not succeed (and thus the component is in Unresolved 
00932       //state) it's also not necessary to try to wire the implementing composite 
00933       if(finalComponentState != RuntimeModel::Component::Unresolved)
00934       {
00935          const RuntimeModel::CompositePtr implComposite = 
00936             boost::dynamic_pointer_cast<RuntimeModel::Composite>(component->implementation());
00937          assert(implComposite);
00938 
00939          //if(!wireComposite(implComposite))
00940          // @changed: (oh)08/10/12
00941          // should be only unresolved if wireComposite returns true, not false!
00942          if(wireComposite(implComposite))
00943          {
00945             BOOST_FOREACH(const RuntimeModel::ReferenceBasePtr& reference, component->references())
00946             {
00947                unwireReference(reference);
00948                reference->setState(RuntimeModel::Reference::Unresolved);
00949             }
00950                        
00951             finalComponentState = RuntimeModel::Component::Unresolved;
00952          }
00953          else
00954          {
00955             implComposite->setState(RuntimeModel::Composite::Deployed);
00956          }
00957       }
00958    }
00959 
00960    component->setState(finalComponentState);        
00961 }

00963 void wireReference( const RuntimeModel::ReferenceBasePtr& reference
00964                   , ComponentsSet& leftComponentsToWire
00965                   , SCAWiresSet& availableSCAWires)
00966 {
00967    HLOGGER();
00968    HLOGGER_INFO(reference->name().c_str());
00969 
00970    //For now we assume that already resolved references can't be proccess by wireReference().
00971    //This might change in future once rewiring is supported.
00972    assert(reference->state() != RuntimeModel::Reference::Resolved);
00973    //A reference that is currently being resolved MUST NOT be in Unresolving state 
00974    assert(reference->state() != RuntimeModel::Reference::Unresolving);
00975 
00976    const RuntimeModel::ComponentPtr component = reference->component().lock();
00977    assert(component);
00978    const RuntimeModel::CompositePtr composite = component->composite().lock();
00979    assert(composite); 
00980    //the given composite is used as implementation in case it has a parent component
00981    const bool isCompositeUsedAsImplementation = 
00982       (composite->component().lock().get() != nullptr);
00983    
00984    //In case of a domain-level composite a reference that is currently being resolved MUST NOT
00985    //be resolved twice. In case of a composite used as implementation a promoted reference might 
00986    //have been set to Resolving state by the parent component (because it configured own targets 
00987    //on it) but might still be in Resolving state because now the component contained in the
00988    //implementing composite has the chance to configure further targets on. Consequently:
00989    //If we are processing a domain-level composite it IS an error to reprocess a reference in 
00990    //Resolving state, in case of composites used as implementation it's NOT an error.   
00991    assert(!isCompositeUsedAsImplementation ? 
00992       reference->state() != RuntimeModel::Reference::Resolving : true);
00993    
00994    if(isCompositeUsedAsImplementation)
00995    {
00996       //Are there any wires already configured through promotion??
00997       if(!reference->wireableSupport().wires().empty())
00998       {          
00999          bool isNonOverridable = 
01000             reference->scaReference() ? reference->scaReference()->isNonOverridable() 
01001                                        : reference->scaReferenceType()->isNonOverridable();         
01002          if(isNonOverridable == false)
01003          {
01004             //Don't add own targets but take those configured through promotion
01005             reference->setState(RuntimeModel::Reference::Resolved);
01006             return;
01007          }
01008            
01009       } //if(reference->state() == RuntimeModel::Reference::Resolved)
01010    }
01011    
01012    //set the state of all the reference and all the promoted references to Resolving
01014    queue<RuntimeModel::ReferenceBaseWeakPtr> toVisit;
01015    toVisit.push(reference);      
01016    do
01017    {  
01018       const RuntimeModel::ReferenceBasePtr ref = toVisit.front().lock();
01019       assert(ref);      
01020       toVisit.pop();
01021       ref->setState(RuntimeModel::Reference::Resolving);      
01022 
01023       BOOST_FOREACH(const RuntimeModel::ReferenceBaseWeakPtr& promotedRef, ref->promotedTo())
01024       {
01025          toVisit.push(promotedRef);
01026       }
01027    }while(!toVisit.empty());
01028    
01029    //copy the wires having this reference as source from availabeSCAWires to myWires and remove
01030    //them from availableSCAWires
01031    vector<SCAModel::WirePtr> myWires;   
01032    SCAWiresSet::const_iterator it = availableSCAWires.begin();
01033    while(it != availableSCAWires.end())
01034    {
01035       string sourceComponentName;
01036       string sourceReferenceName ;
01037       RuntimeModel::CommonFuncs::parseSCAWiringURI( (*it)->source()
01038                                                   , sourceComponentName
01039                                                   , sourceReferenceName );      
01040       //Is this the reference's parent component?
01041       if(sourceComponentName == component->scaComponent()->name())
01042       {
01043          if(sourceReferenceName.empty())
01044          {
01045             //The reference name can be omitted if the source component only has one reference
01046             //Assembly Spec 1.1 line 1876 f.
01047             if(component->references().size() == 1)
01048             {               
01049                //If the parent component has only one reference, then this must be the reference
01050                //that is currently processed, so put it in myWires
01051                myWires.push_back(*it);                
01052             }
01053             else
01054             {
01055                const string errorMessage = 
01056                   errorMessageComponent("ambiguous references", sourceComponentName) + " " +
01057                   errorMessageReferencedInWire("@source", composite);
01058                logError(errorMessage.c_str()); 
01059             }
01060 
01061             //In both cases the wire can be erased from the availabeSCAWires because it doesn't
01062             //have to be processed further outside of this function (either it's a erroneous wire
01063             //and the error has already been issued in the else-case or it will be processed as 
01064             //part of myWires 
01065             availableSCAWires.erase(it++);
01066          }
01067          else
01068          {
01069             //Is the current reference the source of the wire?
01070             if(sourceReferenceName == reference->scaReferenceType()->name())  
01071             {
01072                myWires.push_back(*it);
01073                availableSCAWires.erase(it++);
01074             }
01075             else
01076             {
01077                ++it;               
01078             }   
01079          } //if(sourceReferenceName.empty()){...}else{...}                 
01080       } //if(componentName == component->scaComponent()->name())
01081       else
01082       {            
01083          //The wire was not copied to myWires, so no need for erasing it, 
01084          //just process the next one
01085          ++it;
01086       }
01087    } //while(it != availableSCAWires.end())   
01088 
01089       
01090    //Put all the targets from the <reference>/@target attribute ((Assembly Spec 1.1 line 939)
01091    //as well as from the @target attributes of all the <wire> elements pointing to the
01092    //current reference (Assembly Spec 1.1. lines 1876) in one container to be able to 
01093    //process them homogeneously (in one loop) without  duplicating code .   
01094    
01095    //First copy the targets from <reference>/@target
01096    vector<string> targets(reference->scaReference()->targets());
01097    //now append the targets from <wire>/@target
01098    transform( myWires.begin()
01099             , myWires.end()
01100             , back_inserter(targets)
01101             , boost::bind(&SCAModel::Wire::target, _1));   
01102    assert(targets.size() == reference->scaReference()->targets().size() + myWires.size());
01103       
01104    //In following we assume that there is no mismatch between <reference>/@multiplicity 
01105    //and the actual number of wires defined for this reference because this should have been
01106    //validated at generation time. Let's express this assumption through asserts.
01107    const string multiplicity = 
01108       reference->scaReference() ? reference->scaReference()->multiplicity() 
01109       : reference->scaReferenceType()->multiplicity();
01110    assert( multiplicity == SCAModel::ReferenceMultiplicity0to1 ? targets.size() <= 1 : true);
01111    assert( multiplicity == SCAModel::ReferenceMultiplicity1to1 ? targets.size() == 1 : true);
01112    assert( multiplicity == SCAModel::ReferenceMultiplicity1toN ? targets.size() >= 1 : true);
01113    
01114    //Iterate over all the targets for the reference and try to wire them.   
01115    for(unsigned int i = 0; i < targets.size(); i++)
01116    {
01117       try
01118       {
01119          //this function MUST NOT change the state of the reference
01120          wireReferenceToTarget( reference
01121                               , targets[i]
01122                               , leftComponentsToWire
01123                               , availableSCAWires );  
01124       }
01125       catch(WiringException& e)
01126       {  
01127          //this assumes that all the targets from <reference>/@target are processed before
01128          //all the targets from <wire>/@target 
01129          const string referencedIn = i < reference->scaReference()->targets().size() ? 
01130               errorMessageReferencedInReference(reference, "@target")
01131             : errorMessageReferencedInWire("@target", composite);
01132 
01133          string errorMessage = e.hasServiceInfo() ? 
01134               errorMessageService( e.targetComponentName(), e.targetServiceName(), e.what() )
01135             : errorMessageComponent( e.targetComponentName(), e.what() );
01136          errorMessage += " " + referencedIn;
01137 
01138          logError(errorMessage.c_str());         
01139          HLOGGER_EXCEPTION(errorMessage.c_str());
01140       } //try{...}catch(){...}      
01141    } //for(unsigned int i = 0; i < targets.size(); i++)  
01142    
01143 
01144    if(!reference->wireableSupport().wires().empty())
01145    {
01146       reference->setState(RuntimeModel::Reference::Resolved);
01147    }
01148    else
01149    {
01150       //If the reference couldn't be resolved but is optional (i.e. multiplicity="0..1" or "0..n")
01151       //set the state to OptionalUnresolved.
01152       if(    multiplicity == SCAModel::ReferenceMultiplicity0to1  
01153           || multiplicity == SCAModel::ReferenceMultiplicity0toN ) 
01154       {
01155          reference->setState(RuntimeModel::Reference::OptionalUnresolved);
01156       }
01157       else
01158       {
01159          reference->setState(RuntimeModel::Reference::Unresolved);
01160       }
01161    }
01162 }

01164 void wireReferenceToTarget( const RuntimeModel::ReferenceBasePtr& reference
01165                           , const Trentino::XML::Schema::AnyURI& target
01166                           , ComponentsSet& leftComponentsToWire

01173    assert(component);
01174    const RuntimeModel::CompositePtr composite = component->composite().lock();
01175    assert(composite);

01177    const bool isCompositeUsedAsImplementation = 
01178       (composite->component().lock().get() != nullptr);
01179    const Trentino::Utils::Pattern::StaticClass<Domain> domain;
01180 
01181    string targetComponentName;
01182    string targetServiceName;
01183

01185 
01186    //find target component
01187    RuntimeModel::ComponentPtr targetComponent;
01188    if(isCompositeUsedAsImplementation)
01189    {
01190       //"For a composite used as a component implementation, wires can only link sources
01191       //and targets that are contained in the same composite" 
01192       //(Assembly Spec 1.1 lines 1890 ff. and 980 ff.)
01193       targetComponent = composite->findComponentByName(targetComponentName);   
01194    }
01195    else
01196    {
01197       //For a domain level composite, the search is performed domain-wide.
01198       //Here it's assumed that the components from the current composite already
01199       //have been added to the virtual domain level composite (before the wiring is
01200       //performed). So that we don't need to search additionally separately inside
01201       //the current composite. 
01202       targetComponent = domain->findDomainLevelComponentByName(targetComponentName);
01203    }
01204 
01205    if(!targetComponent)
01206    {  
01207       //implication of [ASM60047]
01208       throw WiringException("component not found", targetComponentName);      
01209    }
01210 
01211    //find target service
01212    assert(targetComponent);
01213    RuntimeModel::ServiceBasePtr targetService;

01216       targetService = targetComponent->findServiceByName(targetServiceName);
01217       if(!targetService)
01218       {
01219          throw WiringException("service not found", targetComponentName, targetServiceName);         
01220       }
01221 
01222       if( !compatible(reference, targetService) )
01223       {
01224          throw WiringException("service incompatible", targetComponentName, targetServiceName);         
01225       }
01226    }
01227    else //if no service name specified

01230       BOOST_FOREACH ( const RuntimeModel::ServiceBasePtr& service
01231          , targetComponent->services() )

01235             //If the component has more than one compatible service, it's an error.
01236             //[ASM60048]

01238             {
01239                throw WiringException("ambiguous services", targetComponentName);
01240             }

01242             targetService = service;            
01243          }
01244       } // BOOST_FOREACH ( const RuntimeModel::ServiceBasePtr& service...
01245 
01246       //If the component has no compatible services, it's an error.
01247       //[ASM60048]
01248       if(!targetService)
01249       {
01250          throw WiringException("no compatible services", targetComponentName);         
01251       }
01252    } //if(!targetServiceName.empty()){...}else{...} 
01253 
01254 
01255    if(targetComponent->state() == RuntimeModel::Component::Initial)
01256    {      
01257       //the component can be in Initial state only if it's part of the current composite 
01258       //and has not been processed so far

01260          leftComponentsToWire.find(targetComponent);
01261       assert(it != leftComponentsToWire.end());

01263       //so let's try to resolve it 
01264       leftComponentsToWire.erase(it);
01265       wireComponent(targetComponent, leftComponentsToWire, availableSCAWires);
01266    }
01267

01269    {
01270       //Now the component could be in unresolved state for the following reasons:
01271       //1) It's not part of the composite currently being deployed (i.e.has been deployed
01272       //   previously as part of another composite) and is left in Udeployed state due to
01273       //   some errors.
01274       //2) It is part of the current composite, has already been processed but left in 
01275       //   Unresovled state due to some errors
01276       if(targetComponent->state() == RuntimeModel::Component::Unresolved)
01277       {
01278          //In any case, we cannot wire to an unresolved component
01279          throw WiringException("component unresolved", targetComponentName);         
01280       }
01281    }
01282    else if(targetComponent->state() == RuntimeModel::Component::Resolving)
01283    {
01284       //We try to wire to a component that is currently being resolved, this 
01285       //is a cyclic dependency (see also RuntimeModel::Component::State::Resolving)
01286       throw WiringException( "Component in 'Resolving' state. " 
01287                              "Cyclic dependencies currently not supported!"
01288                            , targetComponentName);
01289    }
01290 
01291    assert(targetComponent->state() != RuntimeModel::Component::Unresolved);
01292    assert(targetService);
01293    assert(targetService->state() == RuntimeModel::ServiceBase::Available);
01294 
01295    //********************************* ESTABLISH THE WIRE ***********************************
01296    const RuntimeModel::WirePtr theWireBase = RuntimeModel::Wire::create(reference, targetService);
01297    const boost::shared_ptr<RuntimeModel::Wire> theWire = 
01298       boost::shared_dynamic_cast<RuntimeModel::Wire>(theWireBase);
01299    assert(theWire);
01300    //create proxy instance 
01301    const Trentino::Reflection::ReflectObjectPtr proxyReflectObject = 
01302       reference->proxyReflectObject();
01303    assert(proxyReflectObject);
01304    void* theWireRaw = theWire.get();
01305    void* ctrParams[] = {&theWireRaw}; //wire is passed to the proxy instance 
01306    void* const proxyInstance = 
01307       proxyReflectObject->newInstance(
01308          vector<string>(1, "Trentino::RuntimeModel::Wire *"),
01309          ctrParams, 
01310          true );
01311    if(!proxyInstance)
01312    {
01313       throw runtime_error(
01314          string("Could not create service proxy instance of class '")  + 
01315          proxyReflectObject->className() + "' for reference with @name='" + reference->name() + 
01316          "'in component with @name='" +  component->scaComponent()->name() + 
01317          "' in composite with @name='" + composite->scaComposite()->name() );
01318    }
01319 
01321    theWire->setProxy( RuntimeModel::ProxyPtr(proxyInstance, deleterProxy) );
01322    
01323    //create invocation context
01324    const Trentino::Invocation::InvocationContextPtr invocationContext =
01325       boost::make_shared<Trentino::Invocation::InvocationContext>();
01326    invocationContext->setCurrentInvoker(
01327       Trentino::Core::Bootstrapping::RuntimeBootstrapper::instance().invoker());
01328    theWire->setInvocationContext(invocationContext);
01329    
01330    //execute the wiring
01331    theWire->wire(); 
01332 }

01486 void destroyComponentInstances(const RuntimeModel::ComponentPtr& component)
01487 {
01488    assert(component);
01489    assert(component->state() != RuntimeModel::Component::Initial);
01490    //a component that is currently being resolved MUST NOT be unwired
01491    assert(component->state() != RuntimeModel::Component::Resolving);   
01492    
01493    //[1] Set the component's state to Unresolving in order to prevent new invocations to be
01494    //    directed to the component's implementation instances. Currently it's not needed for
01495    //    components implemented by composites, but we do it for state consistency.
01496    if(component->state() != RuntimeModel::Component::Unresolving)
01497    {  
01498       boost::lock_guard<boost::recursive_mutex> componentLock(component->mutex());
01499       component->setState(RuntimeModel::Component::Unresolving);

01501

01503             XSDTypeImplementationComposite )
01504    {

01506             boost::dynamic_pointer_cast<RuntimeModel::Composite>(component->implementation());
01507       assert(implComposite);

01511             XSDTypeImplementationCPP )
01512

01517             componentImplementationManager();
