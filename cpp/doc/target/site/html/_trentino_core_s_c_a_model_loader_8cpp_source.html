00092 void TRENTINO_STDCALL SCAModelLoader::associate( SCAModel::Component* component
00093                                                , SCAModel::ComponentType* componentType)
00094 {
00095    //Since N components can be associated with _one_ componentType (N-to-1 relationship)
00096    //and therefore associate() could be called several times with the same componentType
00097    //argument (associate(c1, ct1); associate(c2, ct1); etc.) it's necessary to check
00098    //whether a shared pointer to the passed componentType already exists and if yes
00099    //just to copy it instead of creating a new shared pointer and thus introducing a bug.  
00100     
00101    BOOST_FOREACH( const SCAModel::ComponentTypeMap::value_type& pair
00102                 , mModelDescriptor.mComponentTypeMap)
00103    {
00104       if(pair.second.get() == componentType)
00105       {
00106          mModelDescriptor.mComponentTypeMap[component] = pair.second;
00107          return;
00108       }
00109    }
00110 
00111    //create a new shared pointer only if it is the first time that associate() is called
00112    //with this componentType argument 
00113    mModelDescriptor.mComponentTypeMap[component] = SCAModel::ComponentTypePtr(componentType);
00114 }
