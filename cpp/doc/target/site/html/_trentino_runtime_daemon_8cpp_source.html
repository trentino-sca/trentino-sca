00080 // Daemonize this process
00081 static void daemonize( const string& lockfile )
00082 {
00083    int lfp = -1;
00084 
00085    // Already a daemon
00086    if ( getppid() == 1 )
00087       return;
00088 
00089    // Create the lock file as the current user
00090    if ( !lockfile.empty() )
00091    {
00092       lfp = open(lockfile.c_str(), O_RDWR|O_CREAT, 0640);
00093 
00094       if ( lfp < 0 )
00095       {
00096          exit(EXIT_FAILURE);
00097       }
00098    }
00099 
00100    // Drop user if there is one, and we were run as root
00101    if ( (getuid() == 0) || (geteuid() == 0) )
00102    {
00103       struct passwd *pw = getpwnam(DaemonUser.c_str());
00104 
00105       if ( pw )
00106       {
00107          setuid( pw->pw_uid );
00108       }
00109    }
00110 
00111    // Trap signals that we expect to recieve
00112    signal(SIGCHLD,child_handler);
00113    signal(SIGUSR1,child_handler);
00114    signal(SIGALRM,child_handler);
00115 
00116    // Fork off the parent process
00117    pid_t pid = fork();
00118 
00119    if ( pid < 0 )
00120    {
00121       exit(EXIT_FAILURE);
00122    }
00123 
00124    // If we got a good PID, then we can exit the parent process.
00125    if ( pid > 0 )
00126    {
00127       // Wait for confirmation from the child via SIGTERM or SIGCHLD, or
00128       // for two seconds to elapse (SIGALRM).  pause() should not return.
00129       alarm(2);
00130       pause();
00131 
00132       exit(EXIT_FAILURE);
00133    }
00134 
00135    // At this point we are executing as the child process
00136    pid_t parent = getppid();
00137 
00138    // Cancel certain signals
00139    signal(SIGCHLD,SIG_DFL); // A child process dies
00140    signal(SIGTSTP,SIG_IGN); // Various TTY signals
00141    signal(SIGTTOU,SIG_IGN);
00142    signal(SIGTTIN,SIG_IGN);
00143    signal(SIGHUP, SIG_IGN); // Ignore hangup signal
00144    signal(SIGTERM,SIG_DFL); // Die on SIGTERM
00145 
00146    // Change the file mode mask
00147    umask(0);
00148 
00149    // Create a new SID for the child process
00150    pid_t sid = setsid();
00151 
00152    if ( sid < 0 )
00153    {
00154       exit(EXIT_FAILURE);
00155    }
00156 
00157    // Change the current working directory.  This prevents the current
00158    // directory from being locked; hence not being able to remove it.
00159    if ( (chdir("/")) < 0 )
00160    {
00161       exit(EXIT_FAILURE);
00162    }
00163 
00164    // Redirect standard files to /dev/null
00165    freopen( "/dev/null", "r", stdin );
00166    freopen( "/dev/null", "w", stdout );
00167    freopen( "/dev/null", "w", stderr );
00168 
00169    // Tell the parent process that we are A-okay
00170    kill( parent, SIGUSR1 );
00171 }
