00064    //attach wire to reference
00065    const RuntimeModel::ReferenceBasePtr reference = mOriginalReference.lock();
00066    assert(reference);  
00067    queue<RuntimeModel::ReferenceBaseWeakPtr> toVisit;
00068    toVisit.push(reference);      
00069    do
00070    {  
00071       const RuntimeModel::ReferenceBasePtr ref = toVisit.front().lock();
00072       assert(ref);      
00073       toVisit.pop();
00074       ref->wireableSupport().attatchWire(shared_from_this());      
00075 
00076       BOOST_FOREACH(const RuntimeModel::ReferenceBaseWeakPtr& promotedRef, ref->promotedTo() )
00077       {
00078          toVisit.push(promotedRef);
00079       }
00080    }while(!toVisit.empty());
00081 
00082    //attach wire to service   
00083    RuntimeModel::ServiceBasePtr service = mOriginalService.lock();  
00084    assert(service);

00086    {
00087       service->wireableSupport().attatchWire(shared_from_this());
