00040 // Sets the current service status and reports it to the SCM.
00041 //
00042 // Parameters:
00043 //   dwCurrentState - The current state (see SERVICE_STATUS)
00044 //   dwWin32ExitCode - The system error code
00045 //   dwWaitHint - Estimated time for pending operation, 
00046 //     in milliseconds
00047 static void reportServiceStatus(DWORD currentState,
00048                                 DWORD win32ExitCode,
00049                                 DWORD waitHint)
00050 {
00051    // Fill in the SERVICE_STATUS structure.
00052    svcStatus.dwCurrentState = currentState;
00053    svcStatus.dwWin32ExitCode = win32ExitCode;
00054    svcStatus.dwWaitHint = waitHint;
00055 
00056    if ( currentState == SERVICE_START_PENDING )
00057    {
00058       svcStatus.dwControlsAccepted = 0;
00059    } else
00060    {
00061       svcStatus.dwControlsAccepted = SERVICE_ACCEPT_STOP;
00062    }
00063 
00064    static DWORD checkPoint = 1;
00065 
00066    if ( (currentState == SERVICE_RUNNING) ||
00067         (currentState == SERVICE_STOPPED) )
00068    {
00069         svcStatus.dwCheckPoint = 0;
00070    } else
00071    {
00072       svcStatus.dwCheckPoint = checkPoint++;
00073    }
00074 
00075    // Report the status of the service to the SCM.
00076    SetServiceStatus( svcStatusHandle, &svcStatus );
00077 }

00179 // Entry point for the service
00180 //
00181 // Parameters:
00182 //   dwArgc - Number of arguments in the lpszArgv array
