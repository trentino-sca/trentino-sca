00062 QName QName::fromString(const string& qNameAsString)
00063 {  
00064    //no namespace name is provided => local part only
00065    if(qNameAsString[0] != '{')
00066    {
00067       return QName(qNameAsString);
00068    }
00069 
00070    const string emptyNamespaceName = "{" + Trentino::XML::Schema::defaultNamespaceName() + "}";
00071    //if qNameAsString starts with emptyNamespaceName
00072    if(!qNameAsString.compare(0, emptyNamespaceName.size(), emptyNamespaceName))
00073    {
00074       throw invalid_argument(
00075          "cannot create QName from \"" + qNameAsString + "\", if namespace name equals to \"" +
00076          Trentino::XML::Schema::defaultNamespaceName() + "\", only the local part should be provided");
00077                
00078    }
00079 
00080    const size_t endOfNamespaceName = qNameAsString.find_first_of('}');         
00081       
00082    //missing closing '}'
00083    if(endOfNamespaceName == string::npos)
00084    {
00085       throw invalid_argument(
00086          "cannot create QName from \"" + qNameAsString + "\", missing closing '}'");
00087    }
00088   
00089    //namespace name as well as the local part provided   
00090    return QName( qNameAsString.substr(endOfNamespaceName + 1)
00091                , qNameAsString.substr(1, endOfNamespaceName - 1) ); 
00092 }

00096 {
00097    if(!mQNameAsString)

00099       boost::shared_ptr<string> tmpQNameAsString;
00100       if(mNamespaceName.empty())
